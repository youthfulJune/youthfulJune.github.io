线性表之链式存储

发表于 2017-05-08 | 分类于 python | 阅读次数
一、线性表的概念
线性表是最基本、最简单、也是最常用的一种数据结构，是某类元素的一个集合，用元素在序列里的位置
和顺序来表示实际应用中的某种有意义的信息，或者表示数据之间的某种关系的元素组。
根据线性表的实际存储方式，分为两种实现模型：
    1.顺序表--将元素顺序的存放在一块连续的存储区里，元素间的顺序关系有它们的存储顺序自然
        表示
    2.链表--将元素存放在通过链接构造起来的一系列存储块中
二、链表
链表是一种常见的基础数据结构，不像顺序表一样连续存储数据，而是在每一个节点(数据存储单元)里
存放下一个节点的位置信息(即地址)；从而解决了顺序表在进行扩充时需要进行数据搬迁的问题
2.1 单链表
单向链表也叫单链表，是链表中最简单的一种形式，它的每个节点包含两个域，一个信息域（元素
域）和一个链接域。这个链接指向链表中的下一个节点，而最后一个节点的链接域则指向一个空值。

class Node():
    '''创建节点'''
    def __init__(self,elem):
        #节点的两个组成部分：元素域和链接域
        self.elem = elem
        self.next = None
#链表类
class SingleLinkList():
    def __init__(self,node=None):
        #链表的指针变量
        self.header = node

    def is_empty(self):
        '''链表是否为空'''
        if self.header == None:
            print("链表中没有元素")
            return True
        else:
            return False

    def length(self):
        '''链表长度'''
        count = 0
        cur = self.header
        while cur != None:
            count += 1
            cur = cur.next
        return count

    def travel(self):
        '''遍历整个链表'''
        cur = self.header
        while cur != None:
            #print(cur.next)
            print(cur.elem)
            cur = cur.next

    def add(self,item):
        '''链表头部添加元素'''
        node = Node(item)
        cur = self.header
        node.next = cur
        self.header = node

    def append(self,item):
        '''链表尾部添加元素'''
        node = Node(item)
        if self.is_empty():
            self.header = node
        else:
            #设置cur变量，使self.header总指向头节点
            cur = self.header
            #循环到达链表尾部
            while cur.next != None:
                cur = cur.next
            cur.next = node

    def insert(self,pos, item):
        '''指定位置添加元素'''
        if pos <= 0:
            self.add(item)
        elif pos > self.length()-1:
            self.append(item)
        else:
            node = Node(item)
            count = 0
            cur = self.header
            while count < (pos-1):
                count += 1
                cur = cur.next
            node.next = cur.next
            cur.next = node

    def remove(self,item):
        '''删除节点'''
        cur = self.header
        forword_cur = None
        while cur != None:
            if cur.elem == item:
                if forword_cur == None:
                    self.header = cur.next
                else:
                    forword_cur.next = cur.next
                break
            else:
                forword_cur = cur
                cur = cur.next

    def search(self,item):
        '''查找节点是否存在'''
        cur = self.header
        while cur != None:
            if cur.elem == item:
                print("已找到节点%s"%item)
            cur = cur.next
2.2 单链表循环
单链表的一个变形是单向循环链表，链表中最后一个节点的next域不再为None，而是指向
链表的头节点。


python实现单链表循环(增、删、改、查)

#节点类
class Node():
    '''创建节点'''
    def __init__(self, elem):
        # 节点的两个组成部分：元素域和链接域
        self.elem = elem
        self.next = None
# 链表类
class SingleLinkList():
    def __init__(self, node=None):
        # 链表的指针变量
        self.header = node

    def is_empty(self):
        '''链表是否为空'''
        if self.header == None:
            print("链表中没有元素")
            return True
        else:
            return False

    def length(self):
        '''链表长度'''
        if self.header == None:
            return 0
        count = 1
        cur = self.header
        while cur.next != self.header:
            count += 1
            cur = cur.next
        return count

    def travel(self):
        '''遍历整个链表'''
        cur = self.header
        print(cur.elem)
        while cur.next != self.header:
            # print(cur.next)
            cur = cur.next
            print(cur.elem)

    def add(self, item):
        '''链表头部添加元素'''
        node = Node(item)
        if self.is_empty():
            self.header = node
            node.next = self.header
        else:
            cur = self.header
            node.next = cur
            while cur.next != self.header:
                cur = cur.next
            self.header = node
            cur.next = self.header

    def append(self, item):
        '''链表尾部添加元素'''
        node = Node(item)
        if self.is_empty():
            self.header = node
            node.next = self.header
        else:
            cur = self.header  # 设置cur变量，使self.header总指向头节点
            # 循环到达链表尾部
            while cur.next != self.header:
                cur = cur.next
            cur.next = node
            node.next = self.header

    def insert(self, pos, item):
        '''指定位置添加元素'''
        if pos <= 0:
            self.add(item)
        elif pos > self.length() - 1:
            self.append(item)
        else:
            node = Node(item)
            count = 0
            cur = self.header
            while count < (pos-1):
                count += 1
                cur = cur.next
            node.next = cur.next
            cur.next = node

    def remove(self, item):
        '''删除节点'''
        if self.is_empty():
            return
        cur = self.header
        forword_cur = None
        while cur.next != self.header:#(此条件已排除单节点和尾节点的情况)
            if cur.elem == item:
                #头节点的情况(若删除的是头节点)
                if cur == self.header:
                    #需要遍历到尾节点
                    tem_cur = cur
                    while tem_cur.next != self.header:
                        tem_cur = tem_cur.next
                    self.header = cur.next
                    tem_cur.next = self.header
                else:
                    forword_cur.next = cur.next

                return
            else:
                forword_cur = cur
                cur = cur.next
        #单节点和尾节点的情况
        if cur.elem == item:
            if cur == self.header:
                self.header = None
            else:
                forword_cur.next = self.header

    def search(self, item):
        '''查找节点是否存在'''
        cur = self.header
        if cur == None:
            print("没有相关节点信息...")
        elif cur.elem == item:
            print("已找到节点%s" % item)

        while cur.next != self.header:
            cur = cur.next
            if cur.elem == item:
                print("已找到节点%s" % item)
2.3 双向链表
每个节点有两个链接：一个指向前一个节点，当此节点为第一个节点时，指向空值；而另一个指向
下一个节点，当此节点为最后一个节点时，指向空值。


python实现双向链表(增、删、改、查)

class Node():
    '''创建节点'''
    def __init__(self,elem):
        #节点的两个组成部分：元素域和链接域
        self.elem = elem
        self.next = None
        self.prev = None

#链表类
class SingleLinkList():
    def __init__(self,node=None):
        #链表的指针变量
        self.header = node

    def is_empty(self):
        '''链表是否为空'''
        if self.header == None:
            return True
        else:
            return False

    def length(self):
        '''链表长度'''
        cur = self.header
        count = 0
        while cur != None:
            count += 1
            cur = cur.next
        return count

    def travel(self):
        '''遍历整个链表'''
        if self.is_empty():
            pass
        cur = self.header
        while cur != None:
            print(cur.elem)
            cur = cur.next

    def add(self,item):
        '''链表头部添加元素'''
        node = Node(item)
        cur = self.header
        if self.is_empty():
            self.header = node
        else:
            self.header = node
            node.next = cur
            cur.prev = node

    def append(self,item):
        '''链表尾部添加元素'''
        node = Node(item)
        if self.is_empty():
            self.header = node
        else:
            cur = self.header
            while cur.next != None:
                cur = cur.next
            node.prev = cur
            cur.next = node

    def insert(self,pos, item):
        '''指定位置添加元素'''
        if pos <= 0:
            self.add(item)
        elif pos > self.length() - 1:
            self.append(item)
        else:
            node = Node(item)
            cur = self.header
            count = 0
            while count < pos:
                count += 1
                cur = cur.next
            node.next = cur
            node.prev = cur.prev
            cur.prev.next = node
            cur.prev = node

    def remove(self,item):
        '''删除节点'''
        if self.is_empty():
            return
        cur = self.header
        while cur != None:
            if cur.elem == item:
                if cur == self.header:
                    if cur.next == None:
                        self.header = None
                    else:
                        self.header = cur.next
                        cur.next.prev = None
                else:
                    if cur.next == None:
                        cur.prev.next = None
                    else:
                        cur.prev.next = cur.next
                        cur.next.prev = cur.prev
                break
            else:
                cur = cur.next

    def search(self,item):
        '''查找节点是否存在'''
        if self.is_empty():
            return
        cur = self.header
        while cur != None:
            if cur.elem == item:
                print("已找到节点%d"%item)
            cur = cur.next
2.4 双向链表循环
python实现双向链表循环(增、删、改、查)
#节点类
class Node():
    '''创建节点'''
    def __init__(self,elem):
        #节点的两个组成部分：元素域和链接域
        self.elem = elem
        self.next = None
        self.prev = None

#链表类
class SingleLinkList():
    def __init__(self,node=None):
        #链表的指针变量
        self.header = node

    def is_empty(self):
        '''链表是否为空'''
        if self.header == None:
            return True
        else:
            return False

    def length(self):
        '''链表长度'''
        cur = self.header
        count = 1
        while cur.next != self.header:
            count += 1
            cur = cur.next
        return count

    def travel(self):
        '''遍历整个链表'''
        if self.is_empty():
            pass
        cur = self.header
        print(cur.elem)
        while cur.next != self.header:
            cur = cur.next
            print(cur.elem)

    def add(self,item):
        '''链表头部添加元素'''
        node = Node(item)
        if self.is_empty():
            self.header = node
            node.next = self.header
            node.prev = self.header
        else:
            cur = self.header
            node.next = cur
            cur.prev = node
            while cur.next != self.header:
                cur = cur.next
            self.header = node
            self.header.prev = cur
            cur.next = self.header

    def append(self,item):
        '''链表尾部添加元素'''
        node = Node(item)
        if self.is_empty():
            self.header = node
            node.next = self.header
            node.prev = self.header
        else:
            cur = self.header
            while cur.next != self.header:
                cur = cur.next
            node.prev = cur
            node.next = self.header
            self.header.prev = node
            cur.next = node

    def insert(self,pos, item):
        '''指定位置添加元素'''
        if pos <= 0:
            self.add(item)
        elif pos > self.length()-1:
            self.append(item)
        else:
            node = Node(item)
            cur = self.header
            count = 0
            while count < pos:
                count += 1
                cur = cur.next
            node.next = cur
            node.prev = cur.prev
            cur.prev.next = node
            cur.prev = node

    def remove(self,item):
        '''删除节点'''
        if self.is_empty():
            return
        cur = self.header
        while cur.next != self.header:#单节点和尾节点
            if cur.elem == item:
                #头节点
                if cur == self.header:
                    tem_cur = cur
                    while tem_cur.next != self.header:
                        tem_cur = tem_cur.next
                    tem_cur.next = cur.next
                    cur.next.prev = tem_cur
                    self.header = cur.next
                else:
                    cur.prev.next = cur.next
                    cur.next.prev = cur.prev
                return
            cur = cur.next

        if cur.elem == item:
            if cur.next == self.header:
                cur.prev.next = self.header
                self.header.prev = cur.prev
            else:
                self.header = None

    def search(self,item):
        '''查找节点是否存在'''
        if self.is_empty():
            return
        cur = self.header
        while cur.next != self.header:#操作的主题不一样，则条件会有所不同
            if cur.elem == item:
                print("已找到节点%d"%item)
                return
            cur = cur.next

        if cur.elem == item:
            print("已找到节点%d"%item)
